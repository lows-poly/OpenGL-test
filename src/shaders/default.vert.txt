#version 330 core

layout ( location = 0 ) in vec3 a_position;
layout ( location = 1 ) in vec3 a_colour;

uniform mat4 u_transform;
out vec4 vertex_colour;

void main()
{
	gl_Position = u_transform * vec4( a_position.x, a_position.y, a_position.z,
	                                  1.0f );
	vertex_colour = vec4( a_colour, 1.0 );
}
